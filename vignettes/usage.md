```r
library(texblocks)
library(texPreview)
```
 



```r
x <- '$\\alpha$'
y <- 'cindy'
class(x) <- 'tb'
class(y) <- 'tb'
```
 

```r
x1 <- x+y+x
x1
#> $\alpha$&cindy&$\alpha$
```
 

```r
x1 + x1
#> $\alpha$&cindy&$\alpha$&$\alpha$&cindy&$\alpha$
x1 / x1
#> $\alpha$&cindy&$\alpha$\\
#> $\alpha$&cindy&$\alpha$
x1 - x1
#> $\alpha$&cindy&$\alpha$\\ \hline
#> $\alpha$&cindy&$\alpha$
```
 

```r
x2 <- x1 / x1
x2 + x2
#> $\alpha$&cindy&$\alpha$&$\alpha$&cindy&$\alpha$\\
#> $\alpha$&cindy&$\alpha$&$\alpha$&cindy&$\alpha$
```
 

```r
x3 <- x2/x2
x2 + x3
#> $\alpha$&cindy&$\alpha$&$\alpha$&cindy&$\alpha$\\
#> $\alpha$&cindy&$\alpha$&$\alpha$&cindy&$\alpha$\\
#> &&&$\alpha$&cindy&$\alpha$\\
#> &&&$\alpha$&cindy&$\alpha$
```
